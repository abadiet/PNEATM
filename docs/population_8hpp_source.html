461 template <typename... Args>
462 void Population<Args...>::speciate (unsigned int target, unsigned int maxIterationsReachTarget, double stepThresh, double a, double b, double c, double speciesSizeEvolutionLimit) {
463  logger->info ("Speciation");
464 
465  std::vector<Species<Args...>> tmpspecies;
466  unsigned int nbSpeciesAlive = 0;
467  unsigned int ite = 0;
468 
469  while (ite < maxIterationsReachTarget && nbSpeciesAlive != target) {
470 
471  // init tmpspecies
472  tmpspecies.clear ();
473  tmpspecies = species;
474  nbSpeciesAlive = 0;
475 
476  // reset tmpspecies
477  for (size_t iSpe = 0; iSpe < tmpspecies.size (); iSpe++) {
478  if (!tmpspecies [iSpe].isDead) { // if the species is still alive
479  tmpspecies [iSpe].members.clear ();
480  }
481  }
482 
483  // speciation
484  for (unsigned int genome_id = 0; genome_id < popSize; genome_id++) {
485  size_t itmpspeciesBest;
486  double dstBest = std::numeric_limits<double>::max ();
487  if (tmpspecies.size () > 0) { // if there is at least one species
488 
489  // we search for the closest species
490  double dst;
491  for (size_t itmpspecies = 0; itmpspecies < tmpspecies.size (); itmpspecies++) {
492  if (!tmpspecies [itmpspecies].isDead && (dst = tmpspecies [itmpspecies].distanceWith (genomes [genome_id], a, b, c)) <= dstBest) {
493  // we found a closer one
494  itmpspeciesBest = itmpspecies;
495  dstBest = dst;
496  }
497  }
498 
499  }
500  if (dstBest >= speciationThresh) {
501  // the closest species is too far or there is no species: we create a new one
502  itmpspeciesBest = tmpspecies.size ();
503  tmpspecies.push_back (Species<Args...> ((unsigned int) itmpspeciesBest, genomes [genome_id]->connections, dstType));
504  }
505 
506  tmpspecies [itmpspeciesBest].members.push_back (genome_id);
507  genomes [genome_id]->speciesId = tmpspecies [itmpspeciesBest].id;
508  }
509 
510  // check how many species are still alive
511  for (size_t iSpe = 0; iSpe < tmpspecies.size (); iSpe++) {
512  if (tmpspecies [iSpe].members.size () == 0) {
513  // the species has no member, the species is dead
514  tmpspecies [iSpe].isDead = true;
515  }
516  if (!tmpspecies [iSpe].isDead) { // if the species is still alive
517  nbSpeciesAlive ++;
518  }
519  }
520 
521  // update speciationThresh
522  if (nbSpeciesAlive < target) {
523  speciationThresh -= stepThresh;
524  } else {
525  if (nbSpeciesAlive > target) {
526  speciationThresh += stepThresh;
527  }
528  }
529 
530  ite++;
531  }
532 
533  // Set tmpspecies as species
534  species.clear ();
535  species = tmpspecies;
536 
537  logger->trace ("speciation result in {0} alive species in {1} iteration(s)", nbSpeciesAlive, ite);
538  if ((float) nbSpeciesAlive > (float) target * 1.3f || (float) nbSpeciesAlive < (float) target * 0.7f) {
539  logger->warn ("There is a huge difference between target ({0}) and the current number of species ({1})", target, nbSpeciesAlive);
540  }
541 
542  // update species
543  for (size_t iSpe = 0; iSpe < species.size (); iSpe++) {
544  if (!species [iSpe].isDead) { // if the species is still alive, this also ensure that there is at least one member
545  //species [iSpe].connections = GetWeightedCentroid ((unsigned int) iSpe);
546 
547  size_t leaderId = species [iSpe].members [0];
548  for (size_t i = 1; i < species [iSpe].members.size (); i++) {
549  if (genomes [species [iSpe].members [i]]->fitness >= genomes [leaderId]->fitness) {
550  if (genomes [species [iSpe].members [i]]->fitness > genomes [leaderId]->fitness || Random_Double (0.0, 1.0, true, false) < 0.5) {
551  leaderId = species [iSpe].members [i];
552  }
553  }
554  }
555  species [iSpe].connections = genomes [leaderId]->connections; // species leader is the more fit genome
556  }
557  }
558 
559  // update all the fitness as we now know the species
560  UpdateFitnesses (speciesSizeEvolutionLimit);
561 }

563 template <typename... Args>
564 std::unordered_map <unsigned int, Connection> Population<Args...>::GetWeightedCentroid (unsigned int speciesId) {
565  std::unordered_map <unsigned int, Connection> result;
566 
567  double sumFitness = 0.0;
568 
569  for (size_t i = 0; i < species [speciesId].members.size (); i++) { // for each genome in the species
570  double fitness = genomes [species [speciesId].members [i]]->fitness;
571 
572  for (const Connection& conn : genomes [species [speciesId].members [i]]->connections) { // for each of its connections
573  if (conn.enabled) { // only pay attention to active ones
574  size_t curResConn = 0;
575  size_t result_sz = result.size ();
576  while (curResConn < result_sz && result [curResConn].innovId != conn.innovId) { // while we have not found any corresponding connection in result
577  curResConn++;
578  }
579  if (curResConn >= result_sz) { // there is no corresponding connections, we add it
580  result [curResConn] = conn;
581  result [curResConn].weight = 0.0; // set its weight as null as all the previous genomes doesn't contains it
582  }
583  result [curResConn].weight += conn.weight * fitness; // we add the connection's weight dot the genome's fitness (weighted centroid, check below)
584  }
585  }
586 
587  sumFitness += fitness;
588  }
589 
590  // we divide each weight by sumFitness to have the average (weighted centroid)
591  if (sumFitness > 0.0) {
592  for (std::pair<unsigned int, Connection>& conn : result) {
593  conn.second.weight /= sumFitness;
594  }
595  } else {
596  // null sumFitness
597  for (std::pair<unsigned int, Connection>& conn : result) {
598  conn.second.weight = std::numeric_limits<double>::max ();
599  }
600  }
601 
602  return result;
603 }

605 template <typename... Args>
606 void Population<Args...>::UpdateFitnesses (double speciesSizeEvolutionLimit) {
607  fittergenome_id = 0;
608  avgFitness = 0;
609  avgFitnessAdjusted = 0;
610 
611  // process avgFitness and found fittergenome_id
612  for (unsigned int i = 0; i < popSize; i++) {
613  avgFitness += genomes [i]->fitness;
614 
615  if (genomes [i]->fitness > genomes [fittergenome_id]->fitness) {
616  fittergenome_id = i;
617  }
618  }
619  avgFitness /= (double) popSize;
620 
621  // process avgFitnessAdjusted
622  for (size_t i = 0; i < species.size (); i++) {
623  if (!species [i].isDead) {
624  // process species' sumFitness
625  species [i].sumFitness = 0;
626  for (size_t j = 0; j < species [i].members.size (); j++) {
627  species [i].sumFitness += genomes [species [i].members [j]]->fitness;
628  }
629 
630  // update species' gensSinceImproved
631  if (species [i].sumFitness / (double) species[i].members.size () > species [i].avgFitness) {
632  // the avgFitness of the species has increased
633  species [i].gensSinceImproved = 0;
634  } else {
635  species [i].gensSinceImproved ++;
636  }
637 
638  // process species' avgFitness and avgFitnessAdjusted
639  species [i].avgFitness = species [i].sumFitness / (double) species [i].members.size ();
640  species [i].avgFitnessAdjusted = species [i].avgFitness / (double) species [i].members.size ();
641 
642  avgFitnessAdjusted += species [i].avgFitness;
643  }
644  }
645  avgFitnessAdjusted /= (double) popSize;
646 
647  // process offsprings
648  for (size_t i = 0; i < species.size (); i ++) {
649  if (!species [i].isDead) {
650  if (species [i].gensSinceImproved < threshGensSinceImproved) {
651  // the species can have offsprings
652 
653  double evolutionFactor = species [i].avgFitnessAdjusted / (avgFitnessAdjusted + std::numeric_limits<double>::min ());
654  if (evolutionFactor > speciesSizeEvolutionLimit) evolutionFactor = speciesSizeEvolutionLimit; // we limit the species evolution factor: a species's size cannot skyrocket from few genomes
655 
656  species [i].allowedOffspring = (int) ((double) species [i].members.size () * evolutionFactor); // note that (int) 0.9 == 0.0
657  } else {
658  // the species cannot have offsprings it has not improved for a long time
659  species[i].allowedOffspring = 0;
660  }
661  }
662  }
663 
664  // add satistics to the file
665  if (statsFile.is_open ()) {
666  statsFile << generation << "," << genomes [fittergenome_id]->fitness << "," << avgFitness << "," << avgFitnessAdjusted << ",";
667  for (size_t i = 0; i < species.size () - 1; i ++) {
668  statsFile << species [i].members.size () << ",";
669  }
670  statsFile << species.back ().members.size () << "\n";
671  }
672 }

674 template <typename... Args>
675 void Population<Args...>::crossover (bool elitism, double crossover_rate) {
676  logger->info ("Crossover");
677  std::vector<std::unique_ptr<Genome<Args...>>> newGenomes;
678  newGenomes.reserve (popSize);
679 
680  if (elitism) { // elitism mode on = we conserve during generations the more fit genome
681  newGenomes.push_back (genomes [fittergenome_id]->clone ());
682  }
683 
684  for (unsigned int iSpe = 0; iSpe < (unsigned int) species.size (); iSpe ++) {
685  if (!species [iSpe].isDead) {
686  for (int k = 0; k < species [iSpe].allowedOffspring; k++) {
687 
688  // choose pseudo-randomly a first parent
689  unsigned int iParent1 = SelectParent (iSpe);
690 
691  if (Random_Double (0.0, 1.0, true, false) < crossover_rate && species [iSpe].members.size () > 1) {
692  // choose pseudo-randomly a second parent
693  unsigned int iParent2 = SelectParent (iSpe); // TODO might be the same parent as iParent1: is that an issue?
694 
695  // clone the more fit
696  unsigned int iMainParent;
697  unsigned int iSecondParent;
698  if (genomes [iParent1]->fitness > genomes [iParent2]->fitness) {
699  iMainParent = iParent1;
700  iSecondParent = iParent2;
701  } else {
702  iMainParent = iParent2;
703  iSecondParent = iParent1;
704  }
705 
706  newGenomes.push_back (genomes [iMainParent]->clone ());
707 
708  // connections shared by both of the parents must be randomly wheighted
709  for (const std::pair<const unsigned int, Connection>& connMainParent : genomes [iMainParent]->connections) {
710  for (const std::pair<const unsigned int, Connection>& connSecondParent : genomes [iSecondParent]->connections) {
711  if (connMainParent.second.innovId == connSecondParent.second.innovId) {
712  if (Random_Double (0.0, 1.0, true, false) < 0.5) { // 50 % of chance for each parent, newGenome already have the wheight of MainParent
713  newGenomes.back ()->connections [connMainParent.second.id].weight = connSecondParent.second.weight;
714  }
715  }
716  }
717  }
718  } else {
719  // the genome is kept for the new generation (there is no crossover which emphasize mutation's effect eg exploration)
720  newGenomes.push_back (genomes [iParent1]->clone ());
721  }
722  }
723  }
724  }
725 
726  int previousSize = (int) newGenomes.size();
727  // add genomes if some are missing
728  for (int k = 0; k < (int) popSize - (int) previousSize; k++) {
729  newGenomes.push_back (std::make_unique<Genome<Args...>> (bias_sch, inputs_sch, outputs_sch, hiddens_sch_init, bias_values, resetValues, activationFns, &conn_innov, &node_innov, N_ConnInit, probRecuInit, weightExtremumInit, maxRecuInit, logger));
730  }
731  // or remove some genomes if there is too many genomes
732  for (int k = 0; k < (int) previousSize - (int) popSize; k++) {
733  newGenomes.pop_back ();
734  }
735 
736  // replace the current genomes by the new ones
737  genomes.clear ();
738  genomes = std::move (newGenomes);
739 
740  // reset species members
741  for (size_t i = 0; i < species.size (); i++) {
742  species [i].members.clear ();
743  species [i].isDead = true;
744  }
745  for (unsigned int i = 0; i < popSize; i++) {
746  if (genomes [i]->speciesId > -1) {
747  species [genomes [i]->speciesId].members.push_back (i);
748  species [genomes [i]->speciesId].isDead = false; // empty species will stay to isDead = true

757 template <typename... Args>
758 int Population<Args...>::SelectParent (unsigned int iSpe) {
759  /* Chooses player from the population to return randomly(considering fitness).
760  This works by randomly choosing a value between 0 and the sum of all the fitnesses then go through all the genomes
761  and add their fitness to a running sum and if that sum is greater than the random value generated,
762  that genome is chosen since players with a higher fitness function add more to the running sum then they have a higher chance of being chosen */
763 
764  if (Eq_Double (species [iSpe].sumFitness, 0.0)) {
765  // everyone as a null fitness: we return a random genome
766  return species [iSpe].members [
767  Random_UInt (0, (unsigned int) species [iSpe].members.size () - 1)
768  ];
769  }
770 
771  double randThresh = Random_Double (0.0, species [iSpe].sumFitness, true, false);
772  double runningSum = 0.0;
773  for (size_t i = 0; i < species [iSpe].members.size (); i++) {
774  runningSum += genomes [species [iSpe].members [i]]->fitness;
775  if (runningSum > randThresh) {
776  return species [iSpe].members [i];
777  }
778  }
779  return -1; // impossible
780 }
