110 template <typename... Args>
111 double Species<Args...>::ConventionalNEAT (const std::unique_ptr<Genome<Args...>>& genome, double a, double b, double c) {
112  // get enabled connections and maxInnovId for genome 1
113  unsigned int maxInnovId1 = 0;
114  std::vector<size_t> connEnabled1;
115  for (size_t i = 0; i < genome->connections.size (); i++) {
116  if (genome->connections [i].enabled) {
117  connEnabled1.push_back(i);
118  if (genome->connections [i].innovId > maxInnovId1) {
119  maxInnovId1 = genome->connections [i].innovId;
120  }
121  }
122  }
123 
124  // get enabled connections and maxInnovId for genome 2
125  unsigned int maxInnovId2 = 0;
126  std::vector<size_t> connEnabled2;
127  for (size_t i = 0; i < connections.size (); i++) {
128  if (connections [i].enabled) {
129  connEnabled2.push_back (i);
130  if (connections [i].innovId > maxInnovId2) {
131  maxInnovId2 = connections [i].innovId;
132  }
133  }
134  }
135 
136  unsigned int excessGenes = 0;
137  unsigned int disjointGenes = 0;
138  double sumDiffWeights = 0.0;
139  unsigned int nbCommonGenes = 0;
140 
141  for (size_t i1 = 0; i1 < connEnabled1.size (); i1++) {
142  // for each enabled connection of the first genome
143  if (genome->connections [connEnabled1 [i1]].innovId > maxInnovId2) {
144  // if connection's innovId is over the maximum one of second genome's connections
145  // it is an excess gene
146  excessGenes += 1;
147  } else {
148  size_t i2 = 0;
149 
150  while (i2 < connEnabled2.size () && connections [connEnabled2 [i2]].innovId != genome->connections [connEnabled1 [i1]].innovId) {
151  i2 ++;
152  }
153  if (i2 == connEnabled2.size ()) {
154  // no connection with the same innovation id have been found in the second genome
155  // it is a disjoint gene
156  disjointGenes += 1;
157  } else {
158  // one connection has the same innovation id
159  nbCommonGenes += 1;
160  double diff = connections [connEnabled2 [i2]].weight - genome->connections [connEnabled1 [i1]].weight;
161  if (diff > 0) {
162  sumDiffWeights += diff;
163  } else {
164  sumDiffWeights += -1 * diff;
165  }
166  }
167  }
168  }
169 
170  for (size_t i2 = 0; i2 < connEnabled2.size (); i2++) {
171  // for each enabled connection of the second genome
172  if (connections [connEnabled2 [i2]].innovId > maxInnovId1) {
173  // if connection's innovId is over the maximum one of first genome's connections
174  // it is an excess gene
175  excessGenes += 1;
176  } else {
177  size_t i1 = 0;
178  while (i1 < connEnabled1.size () && connections [connEnabled2 [i2]].innovId != genome->connections [connEnabled1 [i1]].innovId) {
179  i1 ++;
180  }
181  if (i1 == connEnabled1.size ()) {
182  // no connection with the same innovation id have been found in the first genome
183  // it is a disjoint gene
184  disjointGenes += 1;
185  } // else, the weight's difference has already been processed in the previous for loop
186  }
187  }
188 
189  if (nbCommonGenes > 0) {
190  return (
191  (a * (double) excessGenes + b * (double) disjointGenes) / (double) std::max (connEnabled1.size (), connEnabled2.size ())
192  + c * sumDiffWeights / (double) nbCommonGenes
193  );
194  } else {
195  // there is no common genes between genomes
196  // let's return the maximum double as they might be very differents
197  return std::numeric_limits<double>::max ();
198  }
199 }

201 template <typename... Args>
202 double Species<Args...>::Euclidian (const std::unique_ptr<Genome<Args...>>& genome) {
203  double result = 0.0;
204 
205  std::vector<size_t> usedId;
206  for (const Connection& conn : connections) {
207  size_t i = 0;
208  while (i < genome->connections.size () && genome->connections [i].innovId != conn.innovId) {
209  i++;
210  }
211  if (i >= genome->connections.size ()) {
212  // the connection is not in the genome
213  result += conn.weight * conn.weight;
214  } else {
215  usedId.push_back (i);
216  // the leader and the genome share this connection
217  result += (conn.weight - genome->connections [i].weight) * (conn.weight - genome->connections [i].weight);
218  }
219  }
220 
221  for (size_t i = 0; i < genome->connections.size (); i++) {
222  size_t k = 0;
223  while (k < usedId.size () && usedId [k] != i) {
224  k++;
225  }
226  if (k >= usedId.size ()) {
227  // the connection has not been take into account
228  result += genome->connections [i].weight * genome->connections [i].weight;
229  }
230  }
231 
232  return result; // actualy the euclidian distance is equal to the squared root of result: but this has no effect as we are comparing values
233 }
