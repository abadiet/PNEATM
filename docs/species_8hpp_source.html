136 template <typename... Args>
137 double Species<Args...>::ConventionalNEAT (const std::unique_ptr<Genome<Args...>>& genome, double a, double b, double c) {
138  // get enabled connections and maxInnovId for genome 1
139  unsigned int maxInnovId1 = 0;
140  std::vector<size_t> connEnabled1;
141  for (size_t i = 0; i < genome->connections.size (); i++) {
142  if (genome->connections [i].enabled) {
143  connEnabled1.push_back(i);
144  if (genome->connections [i].innovId > maxInnovId1) {
145  maxInnovId1 = genome->connections [i].innovId;
146  }
147  }
148  }
149 
150  // get enabled connections and maxInnovId for genome 2
151  unsigned int maxInnovId2 = 0;
152  std::vector<size_t> connEnabled2;
153  for (size_t i = 0; i < connections.size (); i++) {
154  if (connections [i].enabled) {
155  connEnabled2.push_back (i);
156  if (connections [i].innovId > maxInnovId2) {
157  maxInnovId2 = connections [i].innovId;
158  }
159  }
160  }
161 
162  unsigned int excessGenes = 0;
163  unsigned int disjointGenes = 0;
164  double sumDiffWeights = 0.0;
165  unsigned int nbCommonGenes = 0;
166 
167  for (size_t i1 = 0; i1 < connEnabled1.size (); i1++) {
168  // for each enabled connection of the first genome
169  if (genome->connections [connEnabled1 [i1]].innovId > maxInnovId2) {
170  // if connection's innovId is over the maximum one of second genome's connections
171  // it is an excess gene
172  excessGenes += 1;
173  } else {
174  size_t i2 = 0;
175 
176  while (i2 < connEnabled2.size () && connections [connEnabled2 [i2]].innovId != genome->connections [connEnabled1 [i1]].innovId) {
177  i2 ++;
178  }
179  if (i2 == connEnabled2.size ()) {
180  // no connection with the same innovation id have been found in the second genome
181  // it is a disjoint gene
182  disjointGenes += 1;
183  } else {
184  // one connection has the same innovation id
185  nbCommonGenes += 1;
186  double diff = connections [connEnabled2 [i2]].weight - genome->connections [connEnabled1 [i1]].weight;
187  if (diff > 0) {
188  sumDiffWeights += diff;
189  } else {
190  sumDiffWeights += -1 * diff;
191  }
192  }
193  }
194  }
195 
196  for (size_t i2 = 0; i2 < connEnabled2.size (); i2++) {
197  // for each enabled connection of the second genome
198  if (connections [connEnabled2 [i2]].innovId > maxInnovId1) {
199  // if connection's innovId is over the maximum one of first genome's connections
200  // it is an excess gene
201  excessGenes += 1;
202  } else {
203  size_t i1 = 0;
204  while (i1 < connEnabled1.size () && connections [connEnabled2 [i2]].innovId != genome->connections [connEnabled1 [i1]].innovId) {
205  i1 ++;
206  }
207  if (i1 == connEnabled1.size ()) {
208  // no connection with the same innovation id have been found in the first genome
209  // it is a disjoint gene
210  disjointGenes += 1;
211  } // else, the weight's difference has already been processed in the previous for loop
212  }
213  }
214 
215  if (nbCommonGenes > 0) {
216  return (
217  (a * (double) excessGenes + b * (double) disjointGenes) / (double) std::max (connEnabled1.size (), connEnabled2.size ())
218  + c * sumDiffWeights / (double) nbCommonGenes
219  );
220  } else {
221  // there is no common genes between genomes
222  // let's return the maximum double as they might be very differents
223  return std::numeric_limits<double>::max ();
224  }
225 }

227 template <typename... Args>
228 double Species<Args...>::Euclidian (const std::unique_ptr<Genome<Args...>>& genome) {
229  double result = 0.0;
230 
231  std::vector<size_t> usedId;
232  for (const Connection& conn : connections) {
233  size_t i = 0;
234  while (i < genome->connections.size () && genome->connections [i].innovId != conn.innovId) {
235  i++;
236  }
237  if (i >= genome->connections.size ()) {
238  // the connection is not in the genome
239  result += conn.weight * conn.weight;
240  } else {
241  usedId.push_back (i);
242  // the leader and the genome share this connection
243  result += (conn.weight - genome->connections [i].weight) * (conn.weight - genome->connections [i].weight);
244  }
245  }
246 
247  for (size_t i = 0; i < genome->connections.size (); i++) {
248  size_t k = 0;
249  while (k < usedId.size () && usedId [k] != i) {
250  k++;
251  }
252  if (k >= usedId.size ()) {
253  // the connection has not been take into account
254  result += genome->connections [i].weight * genome->connections [i].weight;
255  }
256  }
257 
258  return result; // actualy the euclidian distance is equal to the squared root of result: but this has no effect as we are comparing values
259 }
