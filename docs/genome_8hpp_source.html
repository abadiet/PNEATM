263  // bias
264  nbBias = 0;
265  for (size_t i = 0; i < bias_sch.size (); i++) {
266  for (size_t k = 0; k < bias_sch [i]; k++) {
267  // get Node<T_in, T_out>
268  nodes.push_back(CreateNode::get<Args...> (i, i));
269

271  nodes.back ()->id = nbBias;

273  nodes.back ()->index_T_in = (unsigned int) i;
274  nodes.back ()->index_T_out = (unsigned int) i;
275  nodes.back ()->innovId = node_innov->getInnovId (

277  nodes.back ()->index_T_out,
278  nodes.back ()->index_activation_fn,
279  RepetitionNodeCheck (nodes.back ()->index_T_in, nodes.back ()->index_T_out, nodes.back ()->index_activation_fn) - 1
280  );
281  nodes.back ()->loadInput (bias_values [i]); // load input now as it will always be the same
282  nodes.back ()->process (); // load output now as it will always be the same
283  // bias node doesn't requires a reset value as they are never changed
284 
285  nbBias ++;
286  }

289  nbInput = 0;
290  for (size_t i = 0; i < inputs_sch.size (); i++) {
291  for (size_t k = 0; k < inputs_sch [i]; k++) {
292  // get Node<T_in, T_out>
293  nodes.push_back(CreateNode::get<Args...> (i, i));
294 
295  // setup the node
296  nodes.back ()->id = nbBias + nbInput;
297  nodes.back ()->layer = 0;
298  nodes.back ()->index_T_in = (unsigned int) i;
299  nodes.back ()->index_T_out = (unsigned int) i;
300  nodes.back ()->innovId = node_innov->getInnovId (
301  nodes.back ()->index_T_in,
302  nodes.back ()->index_T_out,
303  nodes.back ()->index_activation_fn,
304  RepetitionNodeCheck (nodes.back ()->index_T_in, nodes.back ()->index_T_out, nodes.back ()->index_activation_fn) - 1
305  );
306  nodes.back ()->setResetValue (resetValues [i]);
307 
308  nbInput ++;
309  }
310  }
311  // output
312  nbOutput = 0;
313  int outputLayer;
314  if (hiddens_sch_init.size () > 0) {
315  outputLayer = 2;
316  } else {
317  outputLayer = 1;
318  }
319  for (size_t i = 0; i < outputs_sch.size (); i++) {
320  for (size_t k = 0; k < outputs_sch [i]; k++) {

323

325  nodes.back ()->id = nbBias + nbInput + nbOutput;
326  nodes.back ()->layer = outputLayer;
327  nodes.back ()->index_T_in = (unsigned int) i;
328  nodes.back ()->index_T_out = (unsigned int) i;

336

338  }
339  }

346  nodes.push_back(CreateNode::get<Args...> (i, j));
347

349  nodes.back ()->id = nbBias + nbInput + nbOutput + nbHidden;
350  nodes.back ()->layer = 1;
351  nodes.back ()->index_T_in = (unsigned int) i;
352  nodes.back ()->index_T_out = (unsigned int) j;
353  nodes.back ()->index_activation_fn = Random_UInt (0, (unsigned int) activationFns [i][j].size () - 1) + 1;
354  nodes.back ()->setActivationFn (
355  activationFns [i][j][nodes.back ()->index_activation_fn - 1]
356  );
357  nodes.back ()->innovId = node_innov->getInnovId (
358  nodes.back ()->index_T_in,
359  nodes.back ()->index_T_out,
360  nodes.back ()->index_activation_fn,
361  RepetitionNodeCheck (nodes.back ()->index_T_in, nodes.back ()->index_T_out, nodes.back ()->index_activation_fn) - 1
362  );
363  nodes.back ()->setResetValue (resetValues [i]);
364 
365  nbHidden ++;
366  }
367  }

371  unsigned int iConn = 0;
372  while (iConn < N_ConnInit) {
373 
374  // inNodeId and outNodeId
375  unsigned int inNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);
376  unsigned int outNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);
377

381  inNodeRecu = Random_UInt (0, maxRecuInit);
382  }
383  if (CheckNewConnectionValidity (inNodeId, outNodeId, inNodeRecu)) { // we don't care of former connections as there is no disabled connection for now
384  // innovId
385  const unsigned int innov_id = conn_innov->getInnovId (nodes [inNodeId]->innovId, nodes [outNodeId]->innovId, inNodeRecu);
386 
387  // weight
388  const double weight = Random_Double (- weightExtremumInit, weightExtremumInit);
389 
390  connections.push_back(Connection (innov_id, inNodeId, outNodeId, inNodeRecu, weight, true));
391 
392  if (inNodeRecu == 0 && nodes [outNodeId]->layer <= nodes [inNodeId]->layer) {
393  nodes [outNodeId]->layer = nodes [inNodeId]->layer + 1;
394  UpdateLayers (outNodeId);
395  }
396 
397  iConn ++;
398  }
399  }

441 template <typename... Args>
442 void Genome<Args...>::runNetwork() {
443  /* Process all input and output. For that, it "scans" each layer from the inputs to the last hidden's layer to calculate input with already known value. */
444 
445  // reset input
446  for (size_t i = nbBias + nbInput; i < nodes.size(); i++) {
447  nodes [i]->reset ();
448  }
449 
450  // process nodes[*]->output for input/bias nodes
451  for (unsigned int i = 0; i < nbBias + nbInput; i++) {
452  nodes [i]->process ();
453  }
454 
455  int lastLayer = nodes[nbBias + nbInput]->layer;
456 
457  for (int ilayer = 1; ilayer <= lastLayer; ilayer++) {
458  // process nodes[*]->input
459  for (size_t i = 0; i < connections.size (); i++) {
460  if (connections [i].enabled && nodes [connections [i].outNodeId]->layer == ilayer) { // if the connections still exist and is pointing on the current layer
461  if (connections [i].inNodeRecu == 0) {
462  logger->trace ("processing of connection {3}: adding to node{0}'s input value from dot product between node{1}'s output and {2}", connections [i].outNodeId, connections [i].inNodeId, connections [i].weight, i);
463  nodes [connections [i].outNodeId]->AddToInput (
464  nodes [connections [i].inNodeId]->getOutput (),
465  connections [i].weight
466  );
467  } else { // is recurent
468  if (connections[i].inNodeRecu < (unsigned int) prevNodes.size () + 1) {
469  logger->trace ("processing of connection {3}: adding to node{0}'s input value from dot product between node{1}'s output ({4} generation from now) and {2}", connections [i].outNodeId, connections [i].inNodeId, connections [i].weight, i, connections [i].inNodeRecu);
470  nodes [connections[i].outNodeId]->AddToInput (
471  prevNodes [(unsigned int) prevNodes.size () - connections [i].inNodeRecu][connections [i].inNodeId],
472  connections [i].weight
473  );
474  } else {
475  logger->trace ("processing of connection {3}: cannot add to node{0}'s input value from dot product between node{1}'s output ({4} generation from now) and {2} as this value isn't existing yet!", connections [i].outNodeId, connections [i].inNodeId, connections [i].weight, i, connections [i].inNodeRecu);
476  // the input of the connection isn't existing yet!
477  // we consider that the connection isn't existing
478  }
479  }
480  }
481  }
482 
483  // process nodes[*]->output
484  for (size_t i = 0; i < nodes.size (); i++) {
485  if (nodes [i]->layer == ilayer) {
486  nodes [i]->process ();
487  }
488  }
489  }
490 
491  // we have to store previous values
492  logger->trace ("saving previous nodes' outputs");
493  prevNodes.push_back ({});
494  prevNodes.back ().reserve (nodes.size ());
495  for (const std::unique_ptr<NodeBase>& node : nodes) {
496  prevNodes.back ().push_back (node->getOutput ());
497  }
498 
499 }

550 template <typename... Args>
551 bool Genome<Args...>::CheckNewConnectionValidity (unsigned int inNodeId, unsigned int outNodeId, unsigned int inNodeRecu, int* disabled_conn_id) {
552  if (nodes [inNodeId]->index_T_out != nodes [outNodeId]->index_T_in) return false; // connections must link two same objects
553  if (outNodeId < nbBias + nbInput) return false; // connections cannot point to an input node
554 
555  for (size_t i = 0; i < connections.size (); i++) {
556  if (
557  connections[i].inNodeId == inNodeId
558  && connections[i].outNodeId == outNodeId
559  && connections[i].inNodeRecu == inNodeRecu
560  ) {
561  if (connections[i].enabled) {
562  return false; // it is already an enabled connection
563  } else {
564  // it is a disabled connection
565  *disabled_conn_id = (int) i;
566  }
567  }
568  }
569 
570  if (inNodeRecu > 0) {
571  // if it is a recurrent connection, the unique condition is to not be a copy of another one
572  // as a recurrent connection cannot create a circle
573  return true;
574  }
575 
576  // the new connection should not have an output as inNode
577  // because if this is the case, outNode's layer > inNode's one (wich is the maximum layer, it is the outputs one)
578  if (inNodeId >= nbBias + nbInput && inNodeId < nbBias + nbInput + nbOutput) {
579  return false;
580  }
581 
582  // the new connection should not create a circle in the network
583  // because if this is the case, even after updating layers, we can find a connection that cannot be in the regular direction
584  if (CheckNewConnectionCircle (inNodeId, outNodeId)) {
585  return false; // the connection will create a connection's circle in the network
586  }
587 
588  return true; // test passed well: it is a valid connection!
589 }

591 template <typename... Args>
592 bool Genome<Args...>::CheckNewConnectionCircle (unsigned int inNodeId, unsigned int outNodeId) {
593  if (inNodeId == outNodeId) {
594  return true;
595  }
596  for (size_t iConn = 0; iConn < connections.size (); iConn ++) {
597  if (connections [iConn].inNodeId == outNodeId && connections [iConn].inNodeRecu == 0) {
598  if (CheckNewConnectionCircle (inNodeId, connections [iConn].outNodeId)) {
599  return true;
600  }
601  }
602  }
603  return false;

607 template <typename... Args>
608 void Genome<Args...>::MutateWeights (double mutateWeightThresh, double mutateWeightFullChangeThresh, double mutateWeightFactor) {
609  logger->trace ("mutating of weights");
610  for (size_t i = 0; i < connections.size (); i++) {
611  if (Random_Double (0.0f, 1.0f, true, false) < mutateWeightThresh) {
612  if (Random_Double (0.0f, 1.0f, true, false) < mutateWeightFullChangeThresh) {
613  // reset weight
614  logger->trace ("resetting connection{}'s weight", i);
615  connections [i].weight = Random_Double (- weightExtremumInit, weightExtremumInit);
616  } else {
617  // pertub weight
618  logger->trace ("pertubating connection{}'s weight", i);
619  connections [i].weight *= Random_Double (- mutateWeightFactor, mutateWeightFactor);
620  }
621  }
622  }

625 template <typename... Args>
626 bool Genome<Args...>::AddConnection (innovationConn_t* conn_innov, unsigned int maxRecurrency, unsigned int maxIterationsFindConnectionThresh, double reactivateConnectionThresh) { // return true if the process ended well, false in the other case
627  logger->trace ("adding a new connection");
628 
629  // find valid node pair
630  unsigned int iterationNb = 1;
631  unsigned int inNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);
632  unsigned int outNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);
633  unsigned int inNodeRecu = Random_UInt (0, maxRecurrency);
634  int disabled_conn_id = -1;
635  while (
636  iterationNb < maxIterationsFindConnectionThresh
637  && !CheckNewConnectionValidity (inNodeId, outNodeId, inNodeRecu, &disabled_conn_id)
638  ) {
639  inNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);
640  outNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);
641  inNodeRecu = Random_UInt (0, maxRecurrency);
642  iterationNb ++;
643  }
644 
645  if (iterationNb < maxIterationsFindConnectionThresh) { // a valid connection has been found
646  // mutating
647  if (disabled_conn_id >= 0) { // it is a former connection
648  if (Random_Double (0.0f, 1.0f, true, false) < reactivateConnectionThresh) {
649  logger->trace ("connection{} is re-enabled", disabled_conn_id);
650  connections [disabled_conn_id].enabled = true; // former connection is reactivated
651  return true;
652  } else {
653  logger->warn ("process ended well but no connection has been added during Genome<Args...>::AddConnection");
654  return true; // return true even no connection has been change because process ended well
655  }
656  } else {
657  // innovId
658  const unsigned int innov_id = conn_innov->getInnovId (nodes [inNodeId]->innovId, nodes [outNodeId]->innovId, inNodeRecu);
659 
660  // weight
661  const double weight = Random_Double (- weightExtremumInit, weightExtremumInit);
662 
663  connections.push_back (Connection (innov_id, inNodeId, outNodeId, inNodeRecu, weight, true));
664 
665  // update layers
666  if (inNodeRecu == 0) {
667  // the added connection may have an impact on the network
668  if (nodes [inNodeId]->layer >= nodes [outNodeId]->layer) {
669  // it has an impact
670  nodes [outNodeId]->layer = nodes [inNodeId]->layer + 1; // the node is constraint, its layer is the following one
671  UpdateLayers (outNodeId);
672  }
673  }
674 
675  logger->trace ("connection{0} added between node{1} (with a recurrency of {3}) and node{2}", connections.size () - 1, inNodeId, outNodeId, inNodeRecu);
676  return true;
677  }

685 bool Genome<Args...>::AddNode (innovationConn_t* conn_innov, innovationNode_t* node_innov, unsigned int maxIterationsFindConnectionThresh) { // return true = node created, false = nothing created
686  logger->trace ("adding a node");

688  if (connections.size() > 0) { // if there is no connection, we cannot add a node!
689  unsigned int iConn = Random_UInt (0, (unsigned int) connections.size () - 1);
690  unsigned int iterationNb = 0;
691  while (iterationNb < maxIterationsFindConnectionThresh && !connections [iConn].enabled) {
692  iConn = Random_UInt (0, (unsigned int) connections.size () - 1);
693  iterationNb ++;
694  }
695  if (iterationNb < maxIterationsFindConnectionThresh) { // a connection has been found
696  // disable former connection

698 
699  // setup new node
700  const unsigned int newNodeId = (unsigned int) nodes.size ();
701  const unsigned int iT_in = nodes [connections [iConn].inNodeId]->index_T_in;
702  const unsigned int iT_out = nodes [connections [iConn].outNodeId]->index_T_out;

705

707  nodes.push_back(CreateNode::get<Args...> (iT_in, iT_out));
708 
709  // setup the node
710  nodes.back ()->id = newNodeId;
711  nodes.back ()->layer = -1; // no layer for now
712  nodes.back ()->index_T_in = iT_in;
713  nodes.back ()->index_T_out = iT_out;
714  nodes.back ()->index_activation_fn = Random_UInt (0, (unsigned int) activationFns [iT_in][iT_out].size () - 1) + 1;
715  nodes.back ()->setActivationFn (
716  activationFns [iT_in][iT_out][nodes.back ()->index_activation_fn - 1]
717  );
718  nodes.back ()->innovId = node_innov->getInnovId (
719  nodes.back ()->index_T_in,
720  nodes.back ()->index_T_out,
721  nodes.back ()->index_activation_fn,
722  RepetitionNodeCheck (nodes.back ()->index_T_in, nodes.back ()->index_T_out, nodes.back ()->index_activation_fn) - 1
723  );
724  nodes.back ()->setResetValue (resetValues [iT_in]);
725 
726  // build first connection
727  int inNodeId = connections [iConn].inNodeId;
728  int outNodeId = newNodeId;
729  unsigned int inNodeRecu = connections [iConn].inNodeRecu;
730  unsigned int innovId = conn_innov->getInnovId (nodes [inNodeId]->innovId, nodes [outNodeId]->innovId, inNodeRecu);
731  double weight = connections [iConn].weight;
732

734 
735  connections.push_back (Connection (innovId, inNodeId, outNodeId, inNodeRecu, weight, true));
736 
737  // build second connection
738  inNodeId = newNodeId;
739  outNodeId = connections [iConn].outNodeId;
740  inNodeRecu = 0;
741  innovId = conn_innov->getInnovId (nodes [inNodeId]->innovId, nodes [outNodeId]->innovId, inNodeRecu);
742  weight = Random_Double (- weightExtremumInit, weightExtremumInit);
743 
744  logger->trace ("adding connection{0} between node{1} and node{2}", connections.size (), inNodeId, outNodeId);
745 
746  connections.push_back (Connection (innovId, inNodeId, outNodeId, inNodeRecu, weight, true));
747 
748  // update layers
749  if (connections [iConn].inNodeRecu > 0) { // the connection was recurrent, so the layers are not changed
750  if (nodes [connections [iConn].outNodeId]->layer == 1) { // the output node was on the first layer, we cannot set the new node on the layer 0 (reserved for the input): everything has to moved
751  nodes [newNodeId]->layer = 1; // update newNodeId layer
752  nodes [connections [iConn].outNodeId]->layer = 2; // update outNodeId layer
753  UpdateLayers (connections [iConn].outNodeId); // update other layers
754  } else {
755  // we set the new node's layer to the first one because there is no node connected to it on the same network (with a null recurrency)
756  nodes [newNodeId]->layer = 1; // update newNodeId layer
757  }
758  } else { // else, the node is one layer further in the network
759  nodes [newNodeId]->layer = nodes [connections [iConn].inNodeId]->layer + 1; // update newNodeId layer
760  if (nodes [newNodeId]->layer >= nodes [connections [iConn].outNodeId]->layer) {
761  nodes [connections [iConn].outNodeId]->layer = nodes [newNodeId]->layer + 1; // the node is constraint, its layer is the following one
762  UpdateLayers (connections [iConn].outNodeId);
763  }
764  }
765  return true;
766  } else {
767  logger->warn ("maximum iteration threshold to find an active connection has been reached in Genome<Args...>::AddNode: no node is added");
768  return false; // no active connection found
769  }
770  } else {
771  logger->warn ("there is no connection, no node is added in Genome<Args...>::AddNode");
772  return false; // there is no connection, cannot add a node
773  }

905  UpdateLayers_Recursive (nodeId);

918  outputLayer = nodes [i]->layer + 1;
919  }
920  }
921  for (size_t i = nbBias + nbInput; i < nbBias + nbInput + nbOutput; i ++) {
922  // new layer!
