544  }

603 template <typename... Args>
604 void Genome<Args...>::runNetwork() {
605  /* Process all input and output. For that, it "scans" each layer from the inputs to the last hidden's layer to calculate input with already known value. */
606 
607  N_runNetwork++;
608 
609  // reset input
610  for (size_t i = nbBias + nbInput; i < nodes.size(); i++) {
611  nodes [i]->reset (false);
612  }
613 
614  // process nodes[*]->output for input/bias nodes
615  for (unsigned int i = 0; i < nbBias + nbInput; i++) {
616  nodes [i]->process ();
617  }
618 
619  int lastLayer = nodes [nbBias + nbInput]->layer;
620 
621  for (int ilayer = 1; ilayer <= lastLayer; ilayer++) {
622  // process nodes[*]->input
623  for (size_t i = 0; i < connections.size (); i++) {
624  if (connections [i].enabled && nodes [connections [i].outNodeId]->layer == ilayer) { // if the connections still exist and is pointing to the current layer
625  if (connections [i].inNodeRecu < N_runNetwork) {
626 
627  unsigned int depth = connections [i].inNodeRecu;
628  // if the connection's input node is further in the network it has not been processed and so its previous output is the current one for it!
629  if (nodes [connections [i].inNodeId]->layer >= nodes [connections [i].outNodeId]->layer) depth--;
630 
631  nodes [connections [i].outNodeId]->AddToInput (
632  nodes [connections [i].inNodeId]->getOutput (depth),
633  connections [i].weight
634  );
635  } else {
636  // the input of the connection isn't existing yet!
637  // we consider that the connection isn't existing
638  }
639  }
640  }
641 
642  // process nodes[*]->output & store outputs
643  for (size_t i = 0; i < nodes.size (); i++) {
644  if (nodes [i]->layer == ilayer) {
645  nodes [i]->process ();
646  }
647  }
648  }
649 }

703 template <typename... Args>
704 bool Genome<Args...>::CheckNewConnectionValidity (unsigned int inNodeId, unsigned int outNodeId, unsigned int inNodeRecu, int* disabled_conn_id) {
705  if (nodes [inNodeId]->index_T_out != nodes [outNodeId]->index_T_in) return false; // connections must link two same objects
706  if (outNodeId < nbBias + nbInput) return false; // connections cannot point to an input node
707 
708  for (size_t i = 0; i < connections.size (); i++) {
709  if (
710  connections[i].inNodeId == inNodeId
711  && connections[i].outNodeId == outNodeId
712  && connections[i].inNodeRecu == inNodeRecu
713  ) {
714  if (connections[i].enabled) {
715  return false; // it is already an enabled connection
716  } else {
717  // it is a disabled connection
718  *disabled_conn_id = (int) i;
719  }
720  }
721  }
722 
723  if (inNodeRecu > 0) {
724  // if it is a recurrent connection, the unique condition is to not be a copy of another one
725  // as a recurrent connection cannot create a circle
726  return true;
727  }
728 
729  // the new connection should not have an output as inNode
730  // because if this is the case, outNode's layer > inNode's one (wich is the maximum layer, it is the outputs one)
731  if (inNodeId >= nbBias + nbInput && inNodeId < nbBias + nbInput + nbOutput) {
732  return false;
733  }
734 
735  // the new connection should not create a circle in the network
736  // because if this is the case, even after updating layers, we can find a connection that cannot be in the regular direction
737  if (CheckNewConnectionCircle (inNodeId, outNodeId)) {
738  return false; // the connection will create a connection's circle in the network
739  }
740 
741  return true; // test passed well: it is a valid connection!
742 }

744 template <typename... Args>
745 bool Genome<Args...>::CheckNewConnectionCircle (unsigned int inNodeId, unsigned int outNodeId) {
746  if (inNodeId == outNodeId) {
747  return true;
748  }
749  for (size_t iConn = 0; iConn < connections.size (); iConn ++) {
750  if (connections [iConn].inNodeId == outNodeId && connections [iConn].inNodeRecu == 0) {
751  if (CheckNewConnectionCircle (inNodeId, connections [iConn].outNodeId)) {
752  return true;
753  }
754  }
755  }
756  return false;

786 template <typename... Args>
787 bool Genome<Args...>::AddConnection (innovationConn_t* conn_innov, unsigned int maxRecurrency, unsigned int maxIterationsFindConnectionThresh, double reactivateConnectionThresh) {
788  logger->trace ("adding a new connection");
789 
790  // find valid node pair
791  unsigned int iterationNb = 1;
792  unsigned int inNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);
793  unsigned int outNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);
794  unsigned int inNodeRecu = Random_UInt (0, maxRecurrency);
795  int disabled_conn_id = -1;
796  while (
797  iterationNb < maxIterationsFindConnectionThresh
798  && !CheckNewConnectionValidity (inNodeId, outNodeId, inNodeRecu, &disabled_conn_id)
799  ) {
800  inNodeId = Random_UInt (0, (unsigned int) nodes.size() - 1);

915  UpdateLayers (connections [iConn].outNodeId);
916  }
917  }

919  } else {

922  }
923  } else {
